#include<stdio.h>
int main()
{
    //operation for integer
    int a;
    scanf("%d",&a);
    printf("%d\n",a);
    
    
    //operation for float
    float b;
    scanf("%f",&b);
    printf("%f",b);
    
    //operation for character
    char ch;
    scanf("%c",&ch);
    printf("%c",ch);
}
